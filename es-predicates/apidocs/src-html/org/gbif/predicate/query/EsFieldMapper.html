001package org.gbif.predicate.query;
002
003import org.elasticsearch.index.query.RangeQueryBuilder;
004import org.gbif.api.model.common.search.SearchParameter;
005import org.gbif.api.model.predicate.SimplePredicate;
006
007public interface EsFieldMapper<S extends SearchParameter> {
008
009  String getVerbatimFieldName(S searchParameter);
010
011  String getExactMatchFieldName(S searchParameter);
012
013  String getGeoDistanceField();
014
015  String getGeoShapeField();
016
017  boolean isVocabulary(S searchParameter);
018
019  /**
020   * Adds an "is null" filter if the mapper instructs to do it for the specific predicate. Used
021   * mostly in range queries to give specific semantics to null values.
022   */
023  default boolean includeNullInPredicate(SimplePredicate<S> predicate) {
024    return false;
025  }
026
027  /**
028   * Adds an "is null" filter if the mapper instructs to do it for the range query. Used mostly in
029   * range queries to give specific semantics to null values.
030   */
031  default boolean includeNullInRange(S param, RangeQueryBuilder rangeQueryBuilder) {
032    return false;
033  }
034}
