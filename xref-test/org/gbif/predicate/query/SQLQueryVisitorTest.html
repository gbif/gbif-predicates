82    public void testMoreComplexQuery() throws QueryBuildingException {

84      Predicate taxon2 = new EqualsPredicate<>(OccurrenceSearchParameter.TAXON_KEY, "2", false);
85      DisjunctionPredicate taxa = new DisjunctionPredicate(List.of(taxon1, taxon2));

87      Predicate basis =
88          new InPredicate<>(

90              List.of("HUMAN_OBSERVATION", "MACHINE_OBSERVATION"),
91              false);
92  
93      Predicate UK = new EqualsPredicate<>(OccurrenceSearchParameter.COUNTRY, "GB", false);
94      Predicate IE = new EqualsPredicate<>(OccurrenceSearchParameter.COUNTRY, "IE", false);
95      DisjunctionPredicate countries = new DisjunctionPredicate(List.of(UK, IE));
96  
97      Predicate before1989 = new LessThanOrEqualsPredicate<>(OccurrenceSearchParameter.YEAR, "1989");
98      Predicate in2000 = new EqualsPredicate<>(OccurrenceSearchParameter.YEAR, "2000", false);
99      DisjunctionPredicate years = new DisjunctionPredicate(List.of(before1989, in2000));
100 
101     ConjunctionPredicate p = new ConjunctionPredicate(List.of(taxa, basis, countries, years));
102     String where = visitor.buildQuery(p);
103     assertEquals(
104         "((((stringArrayContains(classifications['defaultChecklistKey'], '1', true)) OR (stringArrayContains(classifications['defaultChecklistKey'], '2', true)))) "
105             + "AND ((basisofrecord IN('HUMAN_OBSERVATION', 'MACHINE_OBSERVATION'))) "
106             + "AND ((countrycode IN(\'GB\', \'IE\'))) "
107             + "AND (((year <= 1989) OR (year = 2000))))",
108         where);
