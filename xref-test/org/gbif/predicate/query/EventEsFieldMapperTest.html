13    private String defaultChecklistKey = "defaultChecklistKey";
14  
15    private static final Set<EventSearchParameter> TAXONOMIC_SET =
16        new HashSet<>(
17            Arrays.asList(
18                EventSearchParameter.SCIENTIFIC_NAME,
19                EventSearchParameter.ACCEPTED_TAXON_KEY,
20                EventSearchParameter.TAXON_KEY,
21                EventSearchParameter.IUCN_RED_LIST_CATEGORY,
22                EventSearchParameter.TAXONOMIC_STATUS));
23  
24    private static final Set<EventSearchParameter> VOCABULARY_SET =
25        new HashSet<>(
26            Arrays.asList(
27                EventSearchParameter.HUMBOLDT_TARGET_LIFE_STAGE_SCOPE,
28                EventSearchParameter.HUMBOLDT_TARGET_DEGREE_OF_ESTABLISHMENT_SCOPE));
29  
30    @Override
31    public String getVerbatimFieldName(EventSearchParameter searchParameter) {
32      if (searchParameter.type() == String.class) {
33        return searchParameter.name().toLowerCase() + ".verbatim";
34      } else {
35        return searchParameter.name().toLowerCase();
36      }
37    }
38  
39    @Override
40    public String getExactMatchFieldName(EventSearchParameter searchParameter) {
41      if (searchParameter.type() == String.class) {
42        return searchParameter.name().toLowerCase() + ".keyword";
43      } else {
44        return searchParameter.name().toLowerCase();
45      }
46    }
47  
48    @Override
49    public String getGeoDistanceField() {
50      return "coordinates";
51    }
52  
53    @Override
54    public String getGeoShapeField() {
55      return "scoordinates";
56    }
57  
58    @Override
59    public String getSearchFieldName(EventSearchParameter searchParameter) {
60      return searchParameter.name();
61    }
62  
63    @Override
64    public boolean isVocabulary(EventSearchParameter searchParameter) {
65      return VOCABULARY_SET.contains(searchParameter);
66    }
67  
68    @Override
69    public boolean isTaxonomic(EventSearchParameter searchParameter) {
70      return TAXONOMIC_SET.contains(searchParameter);
71    }
72  
73    @Override
74    public String getChecklistField(String checklistKey, EventSearchParameter searchParameter) {
75  
76      if (searchParameter == EventSearchParameter.SCIENTIFIC_NAME) {
77        return "classifications."
78            + (checklistKey != null ? checklistKey : defaultChecklistKey)
79            + ".usage.name";
80      }
81  
82      return "classifications."
83          + (checklistKey != null ? checklistKey : defaultChecklistKey)
84          + ".taxonKeys";
85    }
86  
87    @Override
88    public String getNestedPath(EventSearchParameter param) {
89      if (param.name().startsWith("HUMBOLDT_")) {
90        return "event.humboldt";
91      }
92      return null;
93    }
