53    @Override
54    public String getVerbatimFieldName(OccurrenceSearchParameter searchParameter) {
55      if (searchParameter.type() == String.class) {
56        return searchParameter.name().toLowerCase() + ".verbatim";
57      } else {
58        return searchParameter.name().toLowerCase();
59      }
60    }
61  
62    @Override
63    public String getExactMatchFieldName(OccurrenceSearchParameter searchParameter) {
64      if (searchParameter.type() == String.class) {
65        return searchParameter.name().toLowerCase() + ".keyword";
66      } else {
67        return searchParameter.name().toLowerCase();
68      }
69    }
70  
71    @Override
72    public String getGeoDistanceField() {
73      return "coordinates";
74    }
75  
76    @Override
77    public String getGeoShapeField() {
78      return "scoordinates";
79    }
80  
81    @Override
82    public boolean isVocabulary(OccurrenceSearchParameter searchParameter) {
83      return VOCABULARY_SET.contains(searchParameter);
84    }
85  
86    @Override
87    public boolean isTaxonomic(OccurrenceSearchParameter searchParameter) {
88      return TAXONOMIC_SET.contains(searchParameter);

90  
91    @Override
92    public String getChecklistField(String checklistKey, OccurrenceSearchParameter searchParameter) {
93  
94      if (searchParameter == OccurrenceSearchParameter.SCIENTIFIC_NAME) {
95        return "classifications."
96            + (checklistKey != null ? checklistKey : defaultChecklistKey)
97            + ".usage.name";
98      }
99  
100     return "classifications."
101         + (checklistKey != null ? checklistKey : defaultChecklistKey)
102         + ".taxonKeys";
103   }
104 
105   @Override
106   public boolean includeNullInPredicate(SimplePredicate<OccurrenceSearchParameter> predicate) {
107     return predicate.getKey() == OccurrenceSearchParameter.DISTANCE_FROM_CENTROID_IN_METERS
108         && (predicate instanceof GreaterThanOrEqualsPredicate
109             || predicate instanceof GreaterThanPredicate);
110   }
