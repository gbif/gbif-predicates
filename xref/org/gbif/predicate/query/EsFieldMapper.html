13  
14    String getVerbatimFieldName(P searchParameter);
15  
16    String getExactMatchFieldName(P searchParameter);
17  
18    String getGeoDistanceField();
19  
20    String getGeoShapeField();
21  
22    String getSearchFieldName(P searchParameter);
23  
24    boolean isVocabulary(P searchParameter);
25  
26    String getChecklistField(String checklistKey, P searchParameter);
27  
28    String getNestedPath(P searchParameter);
29  
30    String getFullTextField();
31  
32    List<FieldSortBuilder> getDefaultSort();
33  
34    default Optional<QueryBuilder> getDefaultFilter() {
35      return Optional.empty();
36    }
37  
38    default boolean isNestedField(P searchParameter) {
39      return !Strings.isNullOrEmpty(getNestedPath(searchParameter));
40    }
41  
42    /**
43     * Returns true if the search parameter is taxonomic related and hence will be determined by which
44     * checklist is in use.
45     *
46     * @param searchParameter
47     * @return true if a taxonomic parameter
48     */
49    boolean isTaxonomic(P searchParameter);
50  
51    /**
52     * Adds an "is null" filter if the mapper instructs to do it for the specific predicate. Used
53     * mostly in range queries to give specific semantics to null values.
54     */
55    default boolean includeNullInPredicate(SimplePredicate<P> predicate) {
56      return false;
57    }
58  
59    /**
60     * Adds an "is null" filter if the mapper instructs to do it for the range query. Used mostly in
61     * range queries to give specific semantics to null values.
62     */
63    default boolean includeNullInRange(P param, RangeQueryBuilder rangeQueryBuilder) {
64      return false;
65    }
66  
67    EsField getEsField(P parameter);
68  
69    EsField getEsFacetField(P parameter);
70  
71    boolean isDateField(EsField esField);
72  
73    EsField getGeoDistanceEsField();
74  
75    EsField getGeoShapeEsField();
76  }
