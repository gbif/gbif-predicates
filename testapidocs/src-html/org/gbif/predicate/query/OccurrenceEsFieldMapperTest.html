001package org.gbif.predicate.query;
002
003import java.util.Arrays;
004import java.util.HashSet;
005import java.util.Set;
006import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
007import org.gbif.api.model.predicate.GreaterThanOrEqualsPredicate;
008import org.gbif.api.model.predicate.GreaterThanPredicate;
009import org.gbif.api.model.predicate.SimplePredicate;
010
011public class OccurrenceEsFieldMapperTest implements EsFieldMapper<OccurrenceSearchParameter> {
012
013  private static final Set<OccurrenceSearchParameter> VOCABULARY_SET =
014      new HashSet<>(
015          Arrays.asList(
016              OccurrenceSearchParameter.LIFE_STAGE,
017              OccurrenceSearchParameter.PATHWAY,
018              OccurrenceSearchParameter.DEGREE_OF_ESTABLISHMENT,
019              OccurrenceSearchParameter.ESTABLISHMENT_MEANS,
020              OccurrenceSearchParameter.EARLIEST_EON_OR_LOWEST_EONOTHEM,
021              OccurrenceSearchParameter.LATEST_EON_OR_HIGHEST_EONOTHEM,
022              OccurrenceSearchParameter.EARLIEST_ERA_OR_LOWEST_ERATHEM,
023              OccurrenceSearchParameter.LATEST_ERA_OR_HIGHEST_ERATHEM,
024              OccurrenceSearchParameter.EARLIEST_PERIOD_OR_LOWEST_SYSTEM,
025              OccurrenceSearchParameter.LATEST_PERIOD_OR_HIGHEST_SYSTEM,
026              OccurrenceSearchParameter.EARLIEST_EPOCH_OR_LOWEST_SERIES,
027              OccurrenceSearchParameter.LATEST_EPOCH_OR_HIGHEST_SERIES,
028              OccurrenceSearchParameter.EARLIEST_AGE_OR_LOWEST_STAGE,
029              OccurrenceSearchParameter.LATEST_AGE_OR_HIGHEST_STAGE));
030
031  @Override
032  public String getVerbatimFieldName(OccurrenceSearchParameter searchParameter) {
033    if (searchParameter.type() == String.class) {
034      return searchParameter.name().toLowerCase() + ".verbatim";
035    } else {
036      return searchParameter.name().toLowerCase();
037    }
038  }
039
040  @Override
041  public String getExactMatchFieldName(OccurrenceSearchParameter searchParameter) {
042    if (searchParameter.type() == String.class) {
043      return searchParameter.name().toLowerCase() + ".keyword";
044    } else {
045      return searchParameter.name().toLowerCase();
046    }
047  }
048
049  @Override
050  public String getGeoDistanceField() {
051    return "coordinates";
052  }
053
054  @Override
055  public String getGeoShapeField() {
056    return "scoordinates";
057  }
058
059  @Override
060  public boolean isVocabulary(OccurrenceSearchParameter searchParameter) {
061    return VOCABULARY_SET.contains(searchParameter);
062  }
063
064  @Override
065  public boolean includeNullInPredicate(SimplePredicate<OccurrenceSearchParameter> predicate) {
066    return predicate.getKey() == OccurrenceSearchParameter.DISTANCE_FROM_CENTROID_IN_METERS
067        && (predicate instanceof GreaterThanOrEqualsPredicate
068            || predicate instanceof GreaterThanPredicate);
069  }
070}
