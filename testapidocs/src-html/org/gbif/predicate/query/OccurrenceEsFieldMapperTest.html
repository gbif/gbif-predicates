001package org.gbif.predicate.query;
002
003import java.util.Arrays;
004import java.util.HashSet;
005import java.util.Set;
006import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
007import org.gbif.api.model.predicate.GreaterThanOrEqualsPredicate;
008import org.gbif.api.model.predicate.GreaterThanPredicate;
009import org.gbif.api.model.predicate.SimplePredicate;
010
011public class OccurrenceEsFieldMapperTest implements EsFieldMapper<OccurrenceSearchParameter> {
012
013  private String defaultChecklistKey = "defaultChecklistKey";
014
015  private static final Set<OccurrenceSearchParameter> TAXONOMIC_SET =
016      new HashSet<>(
017          Arrays.asList(
018              OccurrenceSearchParameter.SCIENTIFIC_NAME,
019              OccurrenceSearchParameter.VERBATIM_SCIENTIFIC_NAME,
020              OccurrenceSearchParameter.ACCEPTED_TAXON_KEY,
021              OccurrenceSearchParameter.ACCEPTED_TAXON_KEY,
022              OccurrenceSearchParameter.TAXON_KEY,
023              OccurrenceSearchParameter.KINGDOM_KEY,
024              OccurrenceSearchParameter.PHYLUM_KEY,
025              OccurrenceSearchParameter.CLASS_KEY,
026              OccurrenceSearchParameter.ORDER_KEY,
027              OccurrenceSearchParameter.FAMILY_KEY,
028              OccurrenceSearchParameter.GENUS_KEY,
029              OccurrenceSearchParameter.SUBGENUS_KEY,
030              OccurrenceSearchParameter.SPECIES_KEY,
031              OccurrenceSearchParameter.IUCN_RED_LIST_CATEGORY,
032              OccurrenceSearchParameter.TAXONOMIC_STATUS,
033              OccurrenceSearchParameter.TAXONOMIC_ISSUE));
034
035  private static final Set<OccurrenceSearchParameter> VOCABULARY_SET =
036      new HashSet<>(
037          Arrays.asList(
038              OccurrenceSearchParameter.LIFE_STAGE,
039              OccurrenceSearchParameter.PATHWAY,
040              OccurrenceSearchParameter.DEGREE_OF_ESTABLISHMENT,
041              OccurrenceSearchParameter.ESTABLISHMENT_MEANS,
042              OccurrenceSearchParameter.EARLIEST_EON_OR_LOWEST_EONOTHEM,
043              OccurrenceSearchParameter.LATEST_EON_OR_HIGHEST_EONOTHEM,
044              OccurrenceSearchParameter.EARLIEST_ERA_OR_LOWEST_ERATHEM,
045              OccurrenceSearchParameter.LATEST_ERA_OR_HIGHEST_ERATHEM,
046              OccurrenceSearchParameter.EARLIEST_PERIOD_OR_LOWEST_SYSTEM,
047              OccurrenceSearchParameter.LATEST_PERIOD_OR_HIGHEST_SYSTEM,
048              OccurrenceSearchParameter.EARLIEST_EPOCH_OR_LOWEST_SERIES,
049              OccurrenceSearchParameter.LATEST_EPOCH_OR_HIGHEST_SERIES,
050              OccurrenceSearchParameter.EARLIEST_AGE_OR_LOWEST_STAGE,
051              OccurrenceSearchParameter.LATEST_AGE_OR_HIGHEST_STAGE));
052
053  @Override
054  public String getVerbatimFieldName(OccurrenceSearchParameter searchParameter) {
055    if (searchParameter.type() == String.class) {
056      return searchParameter.name().toLowerCase() + ".verbatim";
057    } else {
058      return searchParameter.name().toLowerCase();
059    }
060  }
061
062  @Override
063  public String getExactMatchFieldName(OccurrenceSearchParameter searchParameter) {
064    if (searchParameter.type() == String.class) {
065      return searchParameter.name().toLowerCase() + ".keyword";
066    } else {
067      return searchParameter.name().toLowerCase();
068    }
069  }
070
071  @Override
072  public String getGeoDistanceField() {
073    return "coordinates";
074  }
075
076  @Override
077  public String getGeoShapeField() {
078    return "scoordinates";
079  }
080
081  @Override
082  public boolean isVocabulary(OccurrenceSearchParameter searchParameter) {
083    return VOCABULARY_SET.contains(searchParameter);
084  }
085
086  @Override
087  public boolean isTaxonomic(OccurrenceSearchParameter searchParameter) {
088    return TAXONOMIC_SET.contains(searchParameter);
089  }
090
091  @Override
092  public String getChecklistField(String checklistKey, OccurrenceSearchParameter searchParameter) {
093
094    if (searchParameter == OccurrenceSearchParameter.SCIENTIFIC_NAME) {
095      return "classifications."
096          + (checklistKey != null ? checklistKey : defaultChecklistKey)
097          + ".usage.name";
098    }
099
100    return "classifications."
101        + (checklistKey != null ? checklistKey : defaultChecklistKey)
102        + ".taxonKeys";
103  }
104
105  @Override
106  public boolean includeNullInPredicate(SimplePredicate<OccurrenceSearchParameter> predicate) {
107    return predicate.getKey() == OccurrenceSearchParameter.DISTANCE_FROM_CENTROID_IN_METERS
108        && (predicate instanceof GreaterThanOrEqualsPredicate
109            || predicate instanceof GreaterThanPredicate);
110  }
111}
