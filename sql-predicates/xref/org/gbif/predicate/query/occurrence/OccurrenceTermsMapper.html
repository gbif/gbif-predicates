200   private static final Map<SearchParameter, Term> DENORMED_TERMS = Collections.emptyMap();
201 
202   @Override
203   public Term term(SearchParameter searchParameter) {
204     return PARAM_TO_TERM.get(searchParameter);
205   }
206 
207   @Override
208   public boolean isArray(SearchParameter searchParameter) {
209     return ARRAY_STRING_TERMS.containsKey(searchParameter);
210   }
211 
212   @Override
213   public Term getTermArray(SearchParameter searchParameter) {
214     return ARRAY_STRING_TERMS.get(searchParameter);
215   }
216 
217   @Override
218   public boolean isDenormedTerm(SearchParameter searchParameter) {
219     return DENORMED_TERMS.containsKey(searchParameter);
220   }
221 
222   @Override
223   public SearchParameter getDefaultGadmLevel() {
224     return OccurrenceSearchParameter.GADM_LEVEL_0_GID;
225   }
226 
227   @Override
228   public boolean includeNullInPredicate(SimplePredicate<SearchParameter> predicate) {
229     return OccurrenceSearchParameter.DISTANCE_FROM_CENTROID_IN_METERS == predicate.getKey()
230         && (predicate instanceof GreaterThanOrEqualsPredicate
231             || predicate instanceof GreaterThanPredicate);
232   }
