001package org.gbif.predicate.query;
002
003import org.gbif.api.model.common.search.SearchParameter;
004import org.gbif.api.model.predicate.SimplePredicate;
005import org.gbif.dwc.terms.Term;
006
007/** Interface to encapsulate search parameter conversions to terms. */
008public interface SQLTermsMapper<S extends SearchParameter> {
009
010  Term term(S searchParameter);
011
012  boolean isArray(S searchParameter);
013
014  Term getTermArray(S searchParameter);
015
016  boolean isDenormedTerm(S searchParameter);
017
018  S getDefaultGadmLevel();
019
020  /**
021   * Adds an "is null" filter if the mapper instructs to. Used mostly in range queries to give
022   * specific semantics to null values.
023   */
024  default boolean includeNullInPredicate(SimplePredicate<S> predicate) {
025    return false;
026  }
027}
