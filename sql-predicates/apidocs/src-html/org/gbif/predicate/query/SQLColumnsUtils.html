012
013  // reserved hive words
014  public static final ImmutableSet<String> SQL_RESERVED_WORDS =
015      new ImmutableSet.Builder<String>().add("date", "order", "format", "group").build();

017  // prefix for extension columns
018  private static final String EXTENSION_PRE = "ext_";
019
020  private static final Set<? extends Term> EXTENSION_TERMS =
021      Arrays.stream(Extension.values())
022          .map(ext -> TermFactory.instance().findTerm(ext.getRowType()))
023          .collect(Collectors.toSet());
024

030      ImmutableSet.of(DcTerm.modified);
031
032  private static final Set<? extends Term> INTERPRETED_UTC_DATES_MILLISECONDS =

093  private SQLColumnsUtils() {
094    // empty constructor
095  }
096
097  public static String getSQLColumn(Term term) {
098    if (GbifTerm.verbatimScientificName == term) {
099      return "v_" + DwcTerm.scientificName.simpleName().toLowerCase();
100    }
101    String columnName = term.simpleName().toLowerCase();
102    if (SQL_RESERVED_WORDS.contains(columnName)) {
103      return columnName + '_';
104    }
105    return columnName;
106  }
107
108  /** Gets the Hive column name of the term parameter. */
109  public static String getSQLQueryColumn(Term term) {
110    String columnName = getSQLColumn(term);
111    return isVocabulary(term) ? columnName + ".lineage" : columnName;
112  }
113
114  /** Gets the Hive column name of the term parameter. */
115  public static String getSQLValueColumn(Term term) {
116    String columnName = getSQLColumn(term);
117    return isVocabulary(term)
118        ? isSQLArray(term) ? columnName + ".concepts" : columnName + ".concept"
119        : columnName;
120  }
121
122  /** Gets the Hive column name of the extension parameter. */
123  public static String getSQLQueryColumn(Extension extension) {
124    return EXTENSION_PRE + extension.name().toLowerCase();
125  }
126
127  /** Returns the Hive data type of term parameter. */
128  public static String getSQLType(Term term) {
129    if (isInterpretedNumerical(term)) {
130      return "INT";
131    } else if (isInterpretedLocalDateSeconds(term)) {
132      return "BIGINT";
133    } else if (isInterpretedUtcDateSeconds(term)) {
134      return "BIGINT";
135    } else if (isInterpretedUtcDateMilliseconds(term)) {
136      return "BIGINT";
137    } else if (isInterpretedDouble(term)) {
138      return "DOUBLE";
139    } else if (isInterpretedBoolean(term)) {
140      return "BOOLEAN";
141    } else if (isVocabulary(term)) {
142      if (isSQLArray(term)) {
143        return "STRUCT<concepts: ARRAY<STRING>,lineage: ARRAY<STRING>>";
144      } else {
145        return "STRUCT<concept: STRING,lineage: ARRAY<STRING>>";
146      }
147    } else if (isSQLArray(term)) {
148      return "ARRAY<STRING>";
149    } else {
150      return "STRING";
151    }
152  }
153
154  public static boolean isDate(Term term) {
155    return isInterpretedLocalDateSeconds(term)
156        || isInterpretedUtcDateSeconds(term)
157        || isInterpretedUtcDateMilliseconds(term);
158  }
159
160  /** Checks if the term is stored as a Hive array. */
161  public static boolean isSQLArray(Term term) {
162    return GbifTerm.mediaType == term
163        || GbifTerm.issue == term
164        || GbifInternalTerm.networkKey == term
165        || DwcTerm.identifiedByID == term
166        || DwcTerm.recordedByID == term
167        || GbifInternalTerm.dwcaExtension == term
168        || DwcTerm.datasetID == term
169        || DwcTerm.datasetName == term
170        || DwcTerm.typeStatus == term
171        || DwcTerm.otherCatalogNumbers == term
172        || DwcTerm.recordedBy == term
173        || DwcTerm.identifiedBy == term
174        || DwcTerm.preparations == term
175        || DwcTerm.samplingProtocol == term
176        || DwcTerm.georeferencedBy == term
177        || DwcTerm.higherGeography == term
178        || GbifTerm.projectId == term
179        || GbifTerm.lithostratigraphy == term
180        || GbifTerm.biostratigraphy == term;
181  }
182
183  public static boolean isVocabulary(Term term) {
184    return term instanceof Enum && hasTermAnnotation(term, Vocabulary.class);
185  }
186
187  private static boolean hasTermAnnotation(Term term, Class<? extends Annotation> annotation) {
188    try {
189      return term.getClass().getField(((Enum) term).name()).isAnnotationPresent(annotation);
190    } catch (NoSuchFieldException var3) {
191      throw new IllegalArgumentException(var3);
192    }
193  }
194
195  /** @return true if the term is an interpreted local date (timezone not relevant) */
196  public static boolean isInterpretedLocalDateSeconds(Term term) {
197    return INTERPRETED_LOCAL_DATES_SECONDS.contains(term);
198  }
199
200  /** @return true if the term is an interpreted UTC date stored in seconds */
201  public static boolean isInterpretedUtcDateSeconds(Term term) {
202    return INTERPRETED_UTC_DATES_SECONDS.contains(term);
203  }
204
205  /** @return true if the term is an interpreted UTC date stored in milliseconds */
206  public static boolean isInterpretedUtcDateMilliseconds(Term term) {
207    return INTERPRETED_UTC_DATES_MILLISECONDS.contains(term);
208  }
209
210  /** @return true if the term is an interpreted numerical */
211  public static boolean isInterpretedNumerical(Term term) {
212    return INTERPRETED_NUM.contains(term);
213  }
214
215  /** @return true if the term is an interpreted double */
216  public static boolean isInterpretedDouble(Term term) {
217    return INTERPRETED_DOUBLE.contains(term);
218  }
219
220  /** @return true if the term is an interpreted boolean */
221  public static boolean isInterpretedBoolean(Term term) {
222    return INTERPRETED_BOOLEAN.contains(term);
223  }
224
225  /** @return true if the term is a complex type in Hive: array, struct, json, etc. */
226  public static boolean isComplexType(Term term) {
227    return COMPLEX_TYPE.contains(term);
228  }
229
230  public static boolean isExtensionTerm(Term term) {
231    return EXTENSION_TERMS.contains(term);
232  }
