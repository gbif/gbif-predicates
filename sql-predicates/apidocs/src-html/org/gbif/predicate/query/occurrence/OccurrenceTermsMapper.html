202  private static final Map<SearchParameter, Term> DENORMED_TERMS = Collections.emptyMap();
203
204  @Override
205  public Term term(SearchParameter searchParameter) {
206    return PARAM_TO_TERM.get(searchParameter);
207  }
208
209  @Override
210  public boolean isArray(SearchParameter searchParameter) {
211    return ARRAY_STRING_TERMS.containsKey(searchParameter);
212  }
213
214  @Override
215  public Term getTermArray(SearchParameter searchParameter) {
216    return ARRAY_STRING_TERMS.get(searchParameter);
217  }
218
219  @Override
220  public boolean isDenormedTerm(SearchParameter searchParameter) {
221    return DENORMED_TERMS.containsKey(searchParameter);
222  }
223
224  @Override
225  public SearchParameter getDefaultGadmLevel() {
226    return OccurrenceSearchParameter.GADM_LEVEL_0_GID;
227  }
228
229  @Override
230  public boolean includeNullInPredicate(SimplePredicate<SearchParameter> predicate) {
231    return OccurrenceSearchParameter.DISTANCE_FROM_CENTROID_IN_METERS == predicate.getKey()
232        && (predicate instanceof GreaterThanOrEqualsPredicate
233            || predicate instanceof GreaterThanPredicate);
234  }
