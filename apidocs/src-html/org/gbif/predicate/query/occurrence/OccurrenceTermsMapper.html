197  private static final Map<SearchParameter, Term> DENORMED_TERMS = Collections.emptyMap();
198
199  @Override
200  public Term term(SearchParameter searchParameter) {
201    return PARAM_TO_TERM.get(searchParameter);
202  }
203
204  @Override
205  public boolean isArray(SearchParameter searchParameter) {
206    return ARRAY_STRING_TERMS.containsKey(searchParameter);
207  }
208
209  @Override
210  public Term getTermArray(SearchParameter searchParameter) {
211    return ARRAY_STRING_TERMS.get(searchParameter);
212  }
213
214  @Override
215  public boolean isDenormedTerm(SearchParameter searchParameter) {
216    return DENORMED_TERMS.containsKey(searchParameter);
217  }
218
219  @Override
220  public SearchParameter getDefaultGadmLevel() {
221    return OccurrenceSearchParameter.GADM_LEVEL_0_GID;
222  }
223
224  @Override
225  public boolean includeNullInPredicate(SimplePredicate<SearchParameter> predicate) {
226    return OccurrenceSearchParameter.DISTANCE_FROM_CENTROID_IN_METERS == predicate.getKey()
227        && (predicate instanceof GreaterThanOrEqualsPredicate
228            || predicate instanceof GreaterThanPredicate);
229  }
