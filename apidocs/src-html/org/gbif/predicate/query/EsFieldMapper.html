001package org.gbif.predicate.query;
002
003import org.elasticsearch.index.query.RangeQueryBuilder;
004import org.gbif.api.model.common.search.SearchParameter;
005import org.gbif.api.model.predicate.SimplePredicate;
006
007public interface EsFieldMapper<S extends SearchParameter> {
008
009  String getVerbatimFieldName(S searchParameter);
010
011  String getExactMatchFieldName(S searchParameter);
012
013  String getGeoDistanceField();
014
015  String getGeoShapeField();
016
017  boolean isVocabulary(S searchParameter);
018
019  String getChecklistField(String checklistKey, S searchParameter);
020
021  /**
022   * Returns true if the search parameter is taxonomic related and hence will be determined by which
023   * checklist is in use.
024   *
025   * @param searchParameter
026   * @return true if a taxonomic parameter
027   */
028  boolean isTaxonomic(S searchParameter);
029  /**
030   * Adds an "is null" filter if the mapper instructs to do it for the specific predicate. Used
031   * mostly in range queries to give specific semantics to null values.
032   */
033  default boolean includeNullInPredicate(SimplePredicate<S> predicate) {
034    return false;
035  }
036
037  /**
038   * Adds an "is null" filter if the mapper instructs to do it for the range query. Used mostly in
039   * range queries to give specific semantics to null values.
040   */
041  default boolean includeNullInRange(S param, RangeQueryBuilder rangeQueryBuilder) {
042    return false;
043  }
044}
