123  protected String toSQLField(S param, boolean matchCase) {
124    return Optional.ofNullable(term(param))
125        .map(
126            term -> {
127              String sqlCol = SQLColumnsUtils.getSQLQueryColumn(term);
128              if (String.class.isAssignableFrom(param.type())
129                  && (param != OccurrenceSearchParameter.GEOMETRY)
130                  && !matchCase) {
131                return toSQLLower(sqlCol);
132              }
133              return sqlCol;
134            })
135        .orElseThrow(
136            () ->
137                // QueryBuildingException requires an underlying exception
138                new IllegalArgumentException(
139                    "Search parameter " + param.type().getName() + "." + param + " is not mapped"));
140  }

168            () ->
169                // QueryBuildingException requires an underlying exception
170                new IllegalArgumentException(
171                    "Search parameter " + param + " is not mapped to Hive"));

174  /**
175   * Converts a value to the form expected by Hive based on the SearchParameter. Most values pass by
176   * unaltered. Quotes are added for values that need to be quoted, escaping any existing quotes.
177   *
178   * @param param the type of parameter defining the expected type
179   * @param value the original query value
180   * @return the converted value expected by Hive
181   */
182  protected String toSQLValue(S param, String value, boolean matchCase) {
183    if (Enum.class.isAssignableFrom(param.type())) {
184      // all enum parameters are uppercase
185      return '\'' + value.toUpperCase() + '\'';
186    }
187
188    if (Date.class.isAssignableFrom(param.type())) {
189      // use longs for timestamps expressed as ISO dates
190      LocalDate ld = IsoDateParsingUtils.parseDate(value);
191      Instant i = ld.atStartOfDay(ZoneOffset.UTC).toInstant();
192      if (isInterpretedUtcDateMilliseconds(term(param))) {
193        return String.valueOf(i.toEpochMilli());
194      } else {
195        return String.valueOf(i.getEpochSecond());
196      }
197
198    } else if (Number.class.isAssignableFrom(param.type())
199        || Boolean.class.isAssignableFrom(param.type())) {
200      // do not quote numbers
201      return value;
202
203    } else {
204      // quote value, escape existing quotes
205      String strVal = '\'' + value.replaceAll("'", "\\\\'") + '\'';
206      if (String.class.isAssignableFrom(param.type())
207          && !"GEOMETRY".equals(param.name())
208          && !matchCase) {
209        return toSQLLower(strVal);
210      }
211      return strVal;
212    }

215  /**
216   * Translates a valid {@link org.gbif.api.model.occurrence.Download} object and translates it into
217   * a strings that can be used as the <em>WHERE</em> clause for a SQL (hive or spark) download.
218   *
219   * @param predicate to translate
220   * @return WHERE clause
221   */
222  public String buildQuery(Predicate predicate) throws QueryBuildingException {
223    String query = ALL_QUERY;
224    if (predicate != null) { // null predicate means a SELECT ALL
225      builder = new StringBuilder();
226      visit(predicate);
227      query = builder.toString();
228    }
229
230    // Set to null to prevent old StringBuilders hanging around in case this class is reused
231    // somewhere else
232    builder = null;
233    return query;
234  }

240  public void visit(DisjunctionPredicate predicate) throws QueryBuildingException {
241    // See if this disjunction can be simplified into an IN predicate, which is much faster.
242    // We could overcomplicate this:
243    //   A=1 OR A=2 OR B=3 OR B=4 OR C>5 â†’ A IN(1,2) OR B IN (3,4) OR C>5
244    // but that's a very unusual query for us, so we just check for
245    // - EqualsPredicates everywhere
246    // - on the same search parameter.
247
248    boolean useIn = true;
249    Boolean matchCase = null;
250    List<String> values = new ArrayList<>();
251    S parameter = null;
252    String checklistsKey = null;
253
254    for (Predicate subPredicate : predicate.getPredicates()) {
255      if (subPredicate instanceof EqualsPredicate) {
256        EqualsPredicate<S> equalsSubPredicate = (EqualsPredicate<S>) subPredicate;
257        if (parameter == null) {
258          parameter = equalsSubPredicate.getKey();
259          matchCase = equalsSubPredicate.isMatchCase();
260          checklistsKey = equalsSubPredicate.getChecklistKey();
261        } else if (parameter != equalsSubPredicate.getKey()
262            || matchCase != equalsSubPredicate.isMatchCase()
263            || !Objects.equals(checklistsKey, equalsSubPredicate.getChecklistKey())) {
264          useIn = false;
265          break;
266        }
267        values.add(equalsSubPredicate.getValue());
268      } else {
269        useIn = false;
270        break;
271      }
272    }
273
274    if (useIn) {
275      visit(new InPredicate<>(parameter, values, matchCase));
276    } else {
277      visitCompoundPredicate(predicate, DISJUNCTION_OPERATOR);
278    }
279  }

285  /** Supports all parameters incl taxonKey expansion for higher taxa. */
286  public void visit(EqualsPredicate<S> predicate) throws QueryBuildingException {
287    if (predicate.getKey() == OccurrenceSearchParameter.TAXON_KEY) {
288      appendTaxonKeyFilter(predicate);
289    } else if (predicate.getKey() == OccurrenceSearchParameter.GADM_GID) {
290      appendGadmFilterList(GADM_GIDS, predicate.getValue());
291    } else if (predicate.getKey() == OccurrenceSearchParameter.MEDIA_TYPE) {
292      Optional.ofNullable(VocabularyUtils.lookupEnum(predicate.getValue(), MediaType.class))
293          .ifPresent(
294              mediaType ->
295                  builder.append(
296                      String.format(
297                          getArrayFn().apply(GbifTerm.mediaType), mediaType.name(), true)));
298    } else if (predicate.getKey() == OccurrenceSearchParameter.ISSUE) {
299      builder.append(
300          String.format(
301              getArrayFn().apply(GbifTerm.issue), predicate.getValue().toUpperCase(), true));
302    } else if (sqlTermsMapper.isArray(predicate.getKey())) {
303      builder.append(
304          String.format(
305              getArrayFn().apply(sqlTermsMapper.getTermArray(predicate.getKey())),
306              predicate.getValue().replaceAll("'", "\\\\'"),
307              predicate.isMatchCase()));
308    } else if (sqlTermsMapper.isDenormedTerm(predicate.getKey())) {
309      builder
310          .append("(")
311          .append("(")
312          .append(toSQLField(predicate.getKey(), predicate.isMatchCase()))
313          .append(EQUALS_OPERATOR)
314          .append(toSQLValue(predicate.getKey(), predicate.getValue(), predicate.isMatchCase()))
315          .append(") OR (")
316          .append("stringArrayContains(")
317          .append(toSQLDenormField(predicate.getKey(), true))
318          .append(",")
319          .append(toSQLValue(predicate.getKey(), predicate.getValue(), true))
320          .append("))")
321          .append(")");
322    } else if (SQLColumnsUtils.isVocabulary(term(predicate.getKey()))) {
323      builder.append(
324          String.format(getArrayFn().apply(term(predicate.getKey())), predicate.getValue(), false));
325    } else if (Date.class.isAssignableFrom(predicate.getKey().type())) {
326      // Dates may contain a range even for an EqualsPredicate (e.g. "2000" or "2000-02")
327      // The user's query value is inclusive, but the parsed dateRange is exclusive of the
328      // upperBound to allow including the day itself.
329      //
330      // I.e. a predicate value 2000/2005-03 gives a dateRange [2000-01-01,2005-04-01)
331      Range<LocalDate> dateRange = IsoDateParsingUtils.parseDateRange(predicate.getValue());
332
333      if (dateRange.hasLowerBound() || dateRange.hasUpperBound()) {
334        builder.append('(');
335        if (dateRange.hasLowerBound()) {
336          visitSimplePredicate(
337              predicate,
338              GREATER_THAN_EQUALS_OPERATOR,
339              ISO_DATE_FORMATTER.format(dateRange.lowerEndpoint()));
340          if (dateRange.hasUpperBound()) {
341            builder.append(CONJUNCTION_OPERATOR);
342          }
343        }
344        if (dateRange.hasUpperBound()) {
345          visitSimplePredicate(
346              predicate, LESS_THAN_OPERATOR, ISO_DATE_FORMATTER.format(dateRange.upperEndpoint()));
347        }
348        builder.append(')');
349      }
350    } else if (IsoDateInterval.class.isAssignableFrom(predicate.getKey().type())) {
351      // Dates may contain a range even for an EqualsPredicate (e.g. "2000" or "2000-02")
352      // The user's query value is inclusive, but the parsed dateRange is exclusive of the
353      // upperBound to allow including the day itself.
354      //
355      // I.e. a predicate value 2000/2005-03 gives a dateRange [2000-01-01,2005-04-01)
356      Range<LocalDate> dateRange = IsoDateParsingUtils.parseDateRange(predicate.getValue());
357
358      if (dateRange.hasLowerBound() || dateRange.hasUpperBound()) {
359        // all the stuff here
360        builder.append('(');
361        if (dateRange.hasLowerBound()) {
362          SimplePredicate predicateGte =
363              new EqualsPredicate<>(
364                  InternalOccurrenceSearchParameter.EVENT_DATE_GTE,
365                  predicate.getValue(),
366                  predicate.isMatchCase());
367          visitSimplePredicate(
368              predicateGte,
369              GREATER_THAN_EQUALS_OPERATOR,
370              ISO_DATE_FORMATTER.format(dateRange.lowerEndpoint()));
371          if (dateRange.hasUpperBound()) {
372            builder.append(CONJUNCTION_OPERATOR);
373          }
374        }
375        if (dateRange.hasUpperBound()) {
376          SimplePredicate predicateLte =
377              new EqualsPredicate<>(
378                  InternalOccurrenceSearchParameter.EVENT_DATE_LTE,
379                  predicate.getValue(),
380                  predicate.isMatchCase());
381          visitSimplePredicate(
382              predicateLte,
383              LESS_THAN_OPERATOR,
384              ISO_DATE_FORMATTER.format(dateRange.upperEndpoint()));
385        }
386        builder.append(')');
387      }
388    } else if (predicate.getKey() == OccurrenceSearchParameter.GEOLOGICAL_TIME) {
389      if (SearchTypeValidator.isNumericRange(predicate.getValue())) {
390        Range<Double> range = SearchTypeValidator.parseDecimalRange(predicate.getValue());
391        if (range.hasLowerBound()) {
392          builder
393              .append("geologicaltime.gt")
394              .append(GREATER_THAN_EQUALS_OPERATOR)
395              .append(range.lowerEndpoint());
396        }
397        if (range.hasUpperBound()) {
398          if (range.hasLowerBound()) {
399            builder.append(CONJUNCTION_OPERATOR);
400          }
401          builder
402              .append("geologicaltime.lte")
403              .append(LESS_THAN_EQUALS_OPERATOR)
404              .append(range.upperEndpoint());
405        }
406      } else {
407        builder
408            .append(predicate.getValue())
409            .append(GREATER_THAN_OPERATOR)
410            .append("geologicaltime.gt")
411            .append(CONJUNCTION_OPERATOR)
412            .append(predicate.getValue())
413            .append(LESS_THAN_EQUALS_OPERATOR)
414            .append("geologicaltime.lte");
415      }
416    } else {
417      visitSimplePredicate(predicate, EQUALS_OPERATOR);
418    }
419  }

518  public void visit(RangePredicate<S> predicate) throws QueryBuildingException {
519    builder.append("((");
520
521    if (!Objects.isNull(predicate.getValue().getGte())) {
522      visitSimplePredicate(
523          new LessThanOrEqualsPredicate<>(predicate.getKey(), predicate.getValue().getGte()),
524          GREATER_THAN_EQUALS_OPERATOR);
525    } else {
526      visitSimplePredicate(
527          new LessThanPredicate<>(predicate.getKey(), predicate.getValue().getGt()),
528          GREATER_THAN_OPERATOR);
529    }
530
531    builder.append(")");
532    builder.append(CONJUNCTION_OPERATOR);
533    builder.append("(");
534
535    if (!Objects.isNull(predicate.getValue().getLte())) {
536      visitSimplePredicate(
537          new GreaterThanOrEqualsPredicate<>(predicate.getKey(), predicate.getValue().getLte()),
538          LESS_THAN_EQUALS_OPERATOR);
539    } else {
540      visitSimplePredicate(
541          new GreaterThanOrEqualsPredicate<>(predicate.getKey(), predicate.getValue().getLt()),
542          LESS_THAN_OPERATOR);
543    }
544    builder.append("))");
545  }

547  /*
548   * For large disjunctions, IN predicates are around 3Ã— faster than a perfectly balanced tree of
549   * OR predicates, and around 2Ã— faster than a fairly flat OR query.
550   *
551   * With Hive 1.3.0, balancing OR queries should be internal to Hive:
552   *   https://jira.apache.org/jira/browse/HIVE-11398
553   * but it is probably still better to use an IN, which uses a hash table lookup internally:
554   *   https://jira.apache.org/jira/browse/HIVE-11415#comment-14651085
555   */
556  public void visit(InPredicate<S> predicate) throws QueryBuildingException {
557
558    log.info("InPredicate " + predicate);
559
560    boolean isMatchCase = Optional.ofNullable(predicate.isMatchCase()).orElse(Boolean.FALSE);
561
562    if (isSQLArray(predicate.getKey()) || SQLColumnsUtils.isVocabulary(term(predicate.getKey()))) {
563      // Array values must be converted to ORs.
564      builder.append('(');
565      Iterator<String> iterator = predicate.getValues().iterator();
566      while (iterator.hasNext()) {
567        // Use the equals predicate to get the behaviour for array.
568        visit(new EqualsPredicate<S>(predicate.getKey(), iterator.next(), isMatchCase));
569        if (iterator.hasNext()) {
570          builder.append(DISJUNCTION_OPERATOR);
571        }
572      }
573      builder.append(')');
574    } else if (predicate.getKey().name().equals("TAXON_KEY")) {
575      // Taxon keys must be expanded into a disjunction of in predicates
576      appendTaxonKeyFilter(predicate);
577    } else if (predicate.getKey().name().equals("GADM_GID")) {
578      // GADM GIDs must be expanded into a disjunction of in predicates
579      appendGadmGidFilter(predicate.getValues());
580    } else if (predicate.getKey().name().equals("EVENT_DATE")) {
581      // Event dates must be expanded into a disjunction of conjunction predicates (of comparisons)
582      builder.append('(');
583      Iterator<String> iterator = predicate.getValues().iterator();
584      while (iterator.hasNext()) {
585        // Use the equals predicate to get the behaviour for event dates.
586        visit(new EqualsPredicate<S>(predicate.getKey(), iterator.next(), isMatchCase));
587        if (iterator.hasNext()) {
588          builder.append(DISJUNCTION_OPERATOR);
589        }
590      }
591      builder.append(')');
592
593    } else {
594      builder
595          .append('(')
596          .append(toSQLField(predicate.getKey(), isMatchCase))
597          .append(IN_OPERATOR)
598          .append('(');
599      Iterator<String> iterator = predicate.getValues().iterator();
600      while (iterator.hasNext()) {
601        builder.append(toSQLValue(predicate.getKey(), iterator.next(), isMatchCase));
602        if (iterator.hasNext()) {
603          builder.append(", ");
604        }
605      }
606      builder.append(")");
607
608      // this block can be removed in future if we don't need a denormalized
609      // AVRO extension
610      if (sqlTermsMapper.isDenormedTerm(predicate.getKey())) {
611        builder.append(" OR ").append('(');
612
613        Iterator<String> iterator2 = predicate.getValues().iterator();
614        while (iterator2.hasNext()) {
615          builder
616              .append('(')
617              // FIX ME
618              .append("array_contains(")
619              .append(toSQLDenormField(predicate.getKey(), true))
620              .append(',')
621              .append(toSQLValue(predicate.getKey(), iterator2.next(), true))
622              .append(')')
623              .append(')');
624          if (iterator2.hasNext()) {
625            builder.append(" OR ");
626          }
627        }
628        builder.append(")");
629      }
630      builder.append(")");
631    }
632  }

711  public void visit(WithinPredicate within) throws QueryBuildingException {
712    JtsSpatialContextFactory spatialContextFactory = new JtsSpatialContextFactory();
713    spatialContextFactory.normWrapLongitude = true;
714    spatialContextFactory.srid = 4326;
715    spatialContextFactory.datelineRule = DatelineRule.ccwRect;
716
717    WKTReader reader =
718        new WKTReader(spatialContextFactory.newSpatialContext(), spatialContextFactory);
719
720    try {
721      // the geometry must be valid - it was validated in the predicates constructor
722      Shape geometry = reader.parse(within.getGeometry());
723
724      builder.append('(');
725      String withinGeometry;
726
727      // Add an additional filter to a bounding box around any shapes that aren't quadrilaterals, to
728      // speed up the query.
729      if (geometry instanceof JtsGeometry
730          && ((JtsGeometry) geometry).getGeom().getNumPoints() != 5) {
731        Geometry g = ((JtsGeometry) geometry).getGeom();
732        // Use the Spatial4J-fixed geometry; this is split into a multipolygon if it crosses the
733        // antimeridian.
734        withinGeometry = g.toText();
735
736        Rectangle bounds = geometry.getBoundingBox();
737        boundingBox(bounds);
738        builder.append(CONJUNCTION_OPERATOR);
739
740        // A tool (R?) can generate hundreds of tiny areas spread across the globe, all in a single
741        // multipolygon.
742        // Add bounding boxes for these too.
743        // Example: https://www.gbif.org/occurrence/download/0187894-210914110416597
744        if (g instanceof MultiPolygon && g.getNumGeometries() > 2) {
745          builder.append("((");
746          for (int i = 0; i < g.getNumGeometries(); i++) {
747            if (i > 0) {
748              // Too many clauses exceeds Hive's query parsing stack.
749              if (i % 500 == 0) {
750                builder.append(')');
751                builder.append(DISJUNCTION_OPERATOR);
752                builder.append('(');
753              } else {
754                builder.append(DISJUNCTION_OPERATOR);
755              }
756            }
757            Geometry gi = g.getGeometryN(i);
758            Envelope env = gi.getEnvelopeInternal();
759            boundingBox(
760                new RectangleImpl(
761                    env.getMinX(),
762                    env.getMaxX(),
763                    env.getMinY(),
764                    env.getMaxY(),
765                    geometry.getContext()));
766          }
767          builder.append("))");
768          builder.append(CONJUNCTION_OPERATOR);
769        }
770      } else {
771        withinGeometry = within.getGeometry();
772      }
773      builder
774          .append("contains('")
775          .append(withinGeometry)
776          .append("', ")
777          .append(SQLColumnsUtils.getSQLQueryColumn(DwcTerm.decimalLatitude))
778          .append(", ")
779          .append(SQLColumnsUtils.getSQLQueryColumn(DwcTerm.decimalLongitude));
780      // Without the "= TRUE", the expression may evaluate to TRUE or FALSE for all records,
781      // depending
782      // on the data format (ORC, Avro, Parquet, text) of the table (!).
783      // We could not reproduce the issue on our test cluster, so it seems safest to include this.
784      builder.append(") = TRUE").append(')');
785    } catch (Exception e) {
786      throw new QueryBuildingException(e);
787    }
788  }

805  /**
806   * Given a bounding box, generates greater than / lesser than queries using decimalLatitude and
807   * decimalLongitude to form a bounding box.
808   */
809  private void boundingBox(Rectangle bounds) {
810    builder
811        .append('(')
812        // Latitude is easy:
813        .append(SQLColumnsUtils.getSQLQueryColumn(DwcTerm.decimalLatitude))
814        .append(GREATER_THAN_EQUALS_OPERATOR)
815        .append(bounds.getMinY())
816        .append(CONJUNCTION_OPERATOR)
817        .append(SQLColumnsUtils.getSQLQueryColumn(DwcTerm.decimalLatitude))
818        .append(LESS_THAN_EQUALS_OPERATOR)
819        .append(bounds.getMaxY())
820        .append(CONJUNCTION_OPERATOR)
821
822        // Longitude must take account of crossing the antimeridian:
823        .append('(')
824        .append(SQLColumnsUtils.getSQLQueryColumn(DwcTerm.decimalLongitude))
825        .append(GREATER_THAN_EQUALS_OPERATOR)
826        .append(bounds.getMinX());
827    if (bounds.getMinX() < bounds.getMaxX()) {
828      builder.append(CONJUNCTION_OPERATOR);
829    } else {
830      builder.append(DISJUNCTION_OPERATOR);
831    }
832    builder
833        .append(SQLColumnsUtils.getSQLQueryColumn(DwcTerm.decimalLongitude))
834        .append(LESS_THAN_EQUALS_OPERATOR)
835        .append(bounds.getMaxX())
836        .append(')')
837        .append(')');
838  }

840  /**
841   * Builds a list of predicates joined by 'op' statements. The final statement will look like this:
842   *
843   * <p>
844   *
845   * <pre>
846   * ((predicate) op (predicate) ... op (predicate))
847   * </pre>
848   */
849  public void visitCompoundPredicate(CompoundPredicate predicate, String op)
850      throws QueryBuildingException {
851    builder.append('(');
852    Iterator<Predicate> iterator = predicate.getPredicates().iterator();
853    while (iterator.hasNext()) {
854      Predicate subPredicate = iterator.next();
855      builder.append('(');
856      visit(subPredicate);
857      builder.append(')');
858      if (iterator.hasNext()) {
859        builder.append(op);
860      }
861    }
862    builder.append(')');
863  }

865  public void visitSimplePredicate(SimplePredicate<S> predicate, String op)
866      throws QueryBuildingException {
867    if (Number.class.isAssignableFrom(predicate.getKey().type())) {
868      if (SearchTypeValidator.isNumericRange(predicate.getValue())) {
869        if (Integer.class.equals(predicate.getKey().type())) {
870          visit(
871              toIntegerRangePredicate(
872                  SearchTypeValidator.parseIntegerRange(predicate.getValue()), predicate.getKey()));
873        } else {
874          visit(
875              toNumberRangePredicate(
876                  SearchTypeValidator.parseDecimalRange(predicate.getValue()), predicate.getKey()));
877        }
878        return;
879      }
880    }
881
882    if (sqlTermsMapper.includeNullInPredicate(predicate)) {
883      String column = SQLColumnsUtils.getSQLColumn(term(predicate.getKey()));
884      builder
885          .append('(')
886          .append(toSQLField(predicate.getKey(), predicate.isMatchCase()))
887          .append(op)
888          .append(toSQLValue(predicate.getKey(), predicate.getValue(), predicate.isMatchCase()))
889          .append(DISJUNCTION_OPERATOR)
890          .append(
891              isSQLArray(predicate.getKey())
892                  ? String.format(IS_NOT_NULL_ARRAY_OPERATOR, column)
893                  : column + IS_NULL_OPERATOR)
894          .append(')');
895    } else {
896      builder
897          .append(toSQLField(predicate.getKey(), predicate.isMatchCase()))
898          .append(op)
899          .append(toSQLValue(predicate.getKey(), predicate.getValue(), predicate.isMatchCase()));
900    }
901  }

989  private void appendTaxonKeyFilter(InPredicate<S> taxonKeyPredicate) {

992
993    if (taxonKeyPredicate.getChecklistKey() != null) {
994      builder.append('(');
995      boolean first = true;
996      for (String taxonKey : taxonKeys) {
997        if (!first) {
998          builder.append(DISJUNCTION_OPERATOR);

1013

1018          builder.append(DISJUNCTION_OPERATOR);
1019        }
1020        builder
1021            .append(SQLColumnsUtils.getSQLQueryColumn(term))
1022            .append(IN_OPERATOR)
1023            .append("('")
1024            .append(COMMA_JOINER.join(taxonKeys))
1025            .append("')");

1075  /**
1076   * Converts decimal range into a predicate with the form: field >= range.lower AND field <=
1077   * range.upper.
1078   */
1079  private Predicate toNumberRangePredicate(Range<Double> range, S key) {
1080    if (!range.hasLowerBound()) {
1081      return new LessThanOrEqualsPredicate<S>(
1082          key, String.valueOf(range.upperEndpoint().doubleValue()));
1083    }
1084    if (!range.hasUpperBound()) {
1085      return new GreaterThanOrEqualsPredicate<S>(
1086          key, String.valueOf(range.lowerEndpoint().doubleValue()));
1087    }
1088
1089    ImmutableList.Builder<Predicate> predicates =
1090        new ImmutableList.Builder<Predicate>()
1091            .add(
1092                new GreaterThanOrEqualsPredicate<S>(
1093                    key, String.valueOf(range.lowerEndpoint().doubleValue())))
1094            .add(
1095                new LessThanOrEqualsPredicate<S>(
1096                    key, String.valueOf(range.upperEndpoint().doubleValue())));
1097    return new ConjunctionPredicate(predicates.build());
1098  }

1126  private void visit(Object object) throws QueryBuildingException {
1127    Method method = null;
1128    try {
1129      method = getClass().getMethod("visit", new Class[] {object.getClass()});
1130    } catch (NoSuchMethodException e) {
1131      log.warn(
1132          "Visit method could not be found. That means a unknown Predicate has been passed", e);
1133      throw new IllegalArgumentException("Unknown Predicate", e);
1134    }
1135    try {
1136      method.invoke(this, object);
1137    } catch (IllegalAccessException e) {
1138      log.error(
1139          "This error shouldn't occur if all visit methods are public. Probably a programming error",
1140          e);
1141      Throwables.propagate(e);
1142    } catch (InvocationTargetException e) {
1143      log.info("Exception thrown while building the query", e);
1144      throw new QueryBuildingException(e);
1145    }
1146  }
