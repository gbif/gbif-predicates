077/**
078 * The class translates predicates in GBIF download API to equivalent Elasticsearch query requests.
079 * The json string provided by {@link #buildQuery(Predicate)} can be used with _search get requests
080 * of ES index to produce downloads.
081 */
082@RequiredArgsConstructor
083@Slf4j
084public class EsQueryVisitor<S extends SearchParameter> implements QueryVisitor {
085
086  private final EsFieldMapper<S> esFieldMapper;
087
088  private String getExactMatchOrVerbatimField(SimplePredicate<S> predicate) {
089    return predicate.isMatchCase()
090        ? esFieldMapper.getVerbatimFieldName(predicate.getKey())
091        : esFieldMapper.getExactMatchFieldName(predicate.getKey());
092  }
093
094  private String getExactMatchOrVerbatimField(InPredicate<S> predicate) {
095    return Optional.ofNullable(predicate.isMatchCase()).orElse(Boolean.FALSE)
096        ? esFieldMapper.getVerbatimFieldName(predicate.getKey())
097        : esFieldMapper.getExactMatchFieldName(predicate.getKey());
098  }

100  private String parseParamValue(String value, S parameter) {
101    if (Enum.class.isAssignableFrom(parameter.type())
102        && !Country.class.isAssignableFrom(parameter.type())) {
103      return VocabularyUtils.lookup(value, (Class<Enum<?>>) parameter.type())
104          .map(Enum::name)
105          .orElse(null);
106    }
107    if (Boolean.class.isAssignableFrom(parameter.type())) {
108      return value.toLowerCase();
109    }
110    return value;
111  }

113  /**
114   * Translates a valid {@link org.gbif.api.model.occurrence.Download} object and translates it into
115   * a json query that can be used as the <em>body</em> for _search request of ES index.
116   *
117   * @param predicate to translate
118   * @return body clause
119   */
120  @Override
121  public String buildQuery(Predicate predicate) throws QueryBuildingException {
122    return getQueryBuilder(predicate).orElse(QueryBuilders.matchAllQuery()).toString();
123  }

125  /**
126   * Translates a valid {@link org.gbif.api.model.occurrence.Download} object and translates it into
127   * a json query that can be used as the <em>body</em> for _search request of ES index.
128   *
129   * @param predicate to translate
130   * @return body clause
131   */
132  public Optional<QueryBuilder> getQueryBuilder(Predicate predicate) throws QueryBuildingException {
133    if (predicate != null) {
134      BoolQueryBuilder queryBuilder = QueryBuilders.boolQuery();
135      visit(predicate, queryBuilder);
136      return Optional.of(queryBuilder);
137    }
138    return Optional.empty();
139  }

141  /**
142   * handle conjunction predicate
143   *
144   * @param predicate conjunction predicate
145   * @param queryBuilder root query builder
146   */
147  public void visit(ConjunctionPredicate predicate, BoolQueryBuilder queryBuilder)
148      throws QueryBuildingException {
149    // must query structure is equivalent to AND
150    predicate
151        .getPredicates()
152        .forEach(
153            subPredicate -> {
154              try {
155                BoolQueryBuilder mustQueryBuilder = QueryBuilders.boolQuery();
156                visit(subPredicate, mustQueryBuilder);
157                queryBuilder.filter(addNullableFieldPredicate(subPredicate, mustQueryBuilder));
158              } catch (QueryBuildingException ex) {
159                throw new RuntimeException(ex);
160              }
161            });
162  }
163
164  /**
165   * handle disjunction predicate
166   *
167   * @param predicate disjunction predicate
168   * @param queryBuilder root query builder
169   */
170  public void visit(DisjunctionPredicate predicate, BoolQueryBuilder queryBuilder)
171      throws QueryBuildingException {
172    Map<S, List<EqualsPredicate<S>>> equalsPredicatesReplaceableByIn = groupEquals(predicate);
173
174    predicate
175        .getPredicates()
176        .forEach(
177            subPredicate -> {
178              try {
179                if (!isReplaceableByInPredicate(subPredicate, equalsPredicatesReplaceableByIn)) {
180                  BoolQueryBuilder shouldQueryBuilder = QueryBuilders.boolQuery();
181                  visit(subPredicate, shouldQueryBuilder);
182                  queryBuilder.should(addNullableFieldPredicate(subPredicate, shouldQueryBuilder));
183                }
184              } catch (QueryBuildingException ex) {
185                throw new RuntimeException(ex);
186              }
187            });
188    if (!equalsPredicatesReplaceableByIn.isEmpty()) {
189      toInPredicates(equalsPredicatesReplaceableByIn)
190          .forEach(
191              ep ->
192                  queryBuilder
193                      .should()
194                      .add(
195                          QueryBuilders.termsQuery(
196                              getExactMatchOrVerbatimField(ep),
197                              ep.getValues().stream()
198                                  .map(v -> parseParamValue(v, ep.getKey()))
199                                  .collect(Collectors.toList()))));
200    }
201  }
202
203  /** Checks if a predicate has in grouped and can be replaced later by a InPredicate. */
204  private boolean isReplaceableByInPredicate(
205      Predicate predicate, Map<S, List<EqualsPredicate<S>>> equalsPredicatesReplaceableByIn) {
206    if (!equalsPredicatesReplaceableByIn.isEmpty() && predicate instanceof EqualsPredicate) {
207      EqualsPredicate<S> equalsPredicate = (EqualsPredicate<S>) predicate;
208      return equalsPredicatesReplaceableByIn.containsKey(equalsPredicate.getKey())
209          && equalsPredicatesReplaceableByIn
210              .get(equalsPredicate.getKey())
211              .contains(equalsPredicate);
212    }
213    return false;
214  }
215
216  /** Groups all equals predicates by search parameter. */
217  private Map<S, List<EqualsPredicate<S>>> groupEquals(DisjunctionPredicate predicate) {
218    return predicate.getPredicates().stream()
219        .filter(p -> p instanceof EqualsPredicate)
220        .map(p -> (EqualsPredicate<S>) p)
221        .collect(Collectors.groupingBy(EqualsPredicate::getKey))
222        .entrySet()
223        .stream()
224        .filter(e -> e.getValue().size() > 1)
225        .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
226  }
227
228  /** Transforms the grouped EqualsPredicates into InPredicates. */
229  private List<InPredicate<S>> toInPredicates(Map<S, List<EqualsPredicate<S>>> equalPredicates) {
230    return equalPredicates.entrySet().stream()
231        .map(
232            e ->
233                e.getValue().stream()
234                    .collect(Collectors.groupingBy(EqualsPredicate::isMatchCase))
235                    .entrySet()
236                    .stream()
237                    .map(
238                        group ->
239                            new InPredicate<S>(
240                                e.getKey(),
241                                group.getValue().stream()
242                                    .map(EqualsPredicate::getValue)
243                                    .collect(Collectors.toSet()),
244                                group.getKey()))
245                    .collect(Collectors.toList()))
246        .flatMap(List::stream)
247        .collect(Collectors.toList());
248  }

250  /**
251   * handles EqualPredicate
252   *
253   * @param predicate equalPredicate
254   */
255  public void visit(EqualsPredicate<S> predicate, BoolQueryBuilder queryBuilder) {
256    S parameter = predicate.getKey();
257
258    if (Number.class.isAssignableFrom(predicate.getKey().type())) {
259      if (SearchTypeValidator.isNumericRange(predicate.getValue())) {
260        Range<Double> decimalRange = SearchTypeValidator.parseDecimalRange(predicate.getValue());
261        RangeQueryBuilder rqb =
262            QueryBuilders.rangeQuery(esFieldMapper.getExactMatchFieldName(parameter));
263        if (decimalRange.hasLowerBound()) {
264          rqb.gte(decimalRange.lowerEndpoint());
265        }
266        if (decimalRange.hasUpperBound()) {
267          rqb.lte(decimalRange.upperEndpoint());
268        }
269        queryBuilder.filter().add(addNullableFieldPredicate(predicate, rqb));
270        return;
271      }
272    } else if (Date.class.isAssignableFrom(predicate.getKey().type())) {
273      if (SearchTypeValidator.isDateRange(predicate.getValue())) {
274        // The date range is closed-open, so we need lower â‰¤ date < upper.
275        Range<LocalDate> dateRange = IsoDateParsingUtils.parseDateRange(predicate.getValue());
276        RangeQueryBuilder rqb =
277            QueryBuilders.rangeQuery(esFieldMapper.getExactMatchFieldName(parameter));
278        if (dateRange.hasLowerBound()) {
279          rqb.gte(dateRange.lowerEndpoint());
280        }
281        if (dateRange.hasUpperBound()) {
282          rqb.lt(dateRange.upperEndpoint());
283        }
284        // For a match, the occurrence's date range must be entirely within the search query date
285        // range.
286        // i.e. Q:eventDate=1980 will match rec:eventDate=1980-02, but not
287        // rec:eventDate=1980-10-01/1982-02-02.
288        rqb.relation("within");
289        queryBuilder.filter().add(addNullableFieldPredicate(predicate, rqb));
290        return;
291      }
292    } else if (IsoDateInterval.class.isAssignableFrom(predicate.getKey().type())) {
293      Range<LocalDate> dateRange = IsoDateParsingUtils.parseDateRange(predicate.getValue());
294      RangeQueryBuilder rqb =
295          QueryBuilders.rangeQuery(esFieldMapper.getExactMatchFieldName(parameter));
296      if (dateRange.hasLowerBound()) {
297        rqb.gte(dateRange.lowerEndpoint());
298      }
299      if (dateRange.hasUpperBound()) {
300        rqb.lt(dateRange.upperEndpoint());
301      }
302      // For a match, the occurrence's date range must be entirely within the search query date
303      // range.
304      // i.e. Q:eventDate=1980 will match rec:eventDate=1980-02, but not
305      // rec:eventDate=1980-10-01/1982-02-02.
306      rqb.relation("within");
307      queryBuilder.filter().add(addNullableFieldPredicate(predicate, rqb));
308      return;
309    }
310
311    queryBuilder
312        .filter()
313        .add(
314            QueryBuilders.termQuery(
315                getExactMatchOrVerbatimField(predicate),
316                parseParamValue(predicate.getValue(), parameter)));
317  }

319  private Function<String, Object> parseDate =
320      d -> "*".equals(d) ? null : IsoDateParsingUtils.parseDate(d);
321
322  private Function<String, Object> parseInteger = d -> "*".equals(d) ? null : Integer.parseInt(d);
323
324  private Function<String, Object> parseDouble = d -> "*".equals(d) ? null : Double.parseDouble(d);
325
326  public void visit(RangePredicate<S> predicate, BoolQueryBuilder queryBuilder)
327      throws QueryBuildingException {
328
329    RangeQueryBuilder rqb =
330        QueryBuilders.rangeQuery(esFieldMapper.getExactMatchFieldName(predicate.getKey()));
331
332    if (Integer.class.isAssignableFrom(predicate.getKey().type())) {
333      initialiseRangeQuery(predicate, rqb, parseInteger);
334    } else if (Double.class.isAssignableFrom(predicate.getKey().type())) {
335      initialiseRangeQuery(predicate, rqb, parseDouble);
336    } else if (Date.class.isAssignableFrom(predicate.getKey().type())) {
337      initialiseRangeQuery(predicate, rqb, parseDate);
338    }
339
340    queryBuilder.filter().add(rqb);
341  }
342
343  private void initialiseRangeQuery(
344      RangePredicate<S> predicate, RangeQueryBuilder rqb, Function<String, Object> initialiser) {
345    if (predicate.getValue().getGte() != null) {
346      rqb.gte(initialiser.apply(predicate.getValue().getGte()));
347    }
348    if (predicate.getValue().getLte() != null) {
349      rqb.lte(initialiser.apply(predicate.getValue().getLte()));
350    }
351    if (predicate.getValue().getGt() != null) {
352      rqb.gt(initialiser.apply(predicate.getValue().getGt()));
353    }
354    if (predicate.getValue().getLt() != null) {
355      rqb.lt(initialiser.apply(predicate.getValue().getLt()));
356    }
357  }

380  /**
381   * Adds an "is null" filter if the mapper instructs to. Used mostly in range queries to give
382   * specific semantics to null values.
383   */
384  private QueryBuilder addNullableFieldPredicate(Predicate predicate, QueryBuilder filter) {
385    if (predicate instanceof SimplePredicate
386        && esFieldMapper.includeNullInPredicate((SimplePredicate<S>) predicate)) {
387      return QueryBuilders.boolQuery()
388          .should(filter)
389          .should(
390              QueryBuilders.boolQuery()
391                  .mustNot(
392                      QueryBuilders.existsQuery(
393                          esFieldMapper.getExactMatchFieldName(
394                              ((SimplePredicate<S>) predicate).getKey()))));
395    }
396    return filter;
397  }

416  /**
417   * handle IN Predicate
418   *
419   * @param predicate InPredicate
420   * @param queryBuilder root query builder
421   */
422  public void visit(InPredicate<S> predicate, BoolQueryBuilder queryBuilder) {
423    S parameter = predicate.getKey();
424
425    // EVENT_DATE needs special handling
426    if (parameter == OccurrenceSearchParameter.EVENT_DATE) {
427      predicate
428          .getValues()
429          .forEach(
430              value -> {
431                try {
432                  BoolQueryBuilder shouldQueryBuilder = QueryBuilders.boolQuery();
433                  Predicate subPredicate =
434                      new EqualsPredicate<>(
435                          OccurrenceSearchParameter.EVENT_DATE, value, predicate.isMatchCase());
436                  visit(subPredicate, shouldQueryBuilder);
437                  queryBuilder.should(addNullableFieldPredicate(subPredicate, shouldQueryBuilder));
438                } catch (QueryBuildingException ex) {
439                  throw new RuntimeException(ex);
440                }
441              });
442
443    } else {
444      queryBuilder
445          .filter()
446          .add(
447              QueryBuilders.termsQuery(
448                  getExactMatchOrVerbatimField(predicate),
449                  predicate.getValues().stream()
450                      .map(v -> parseParamValue(v, parameter))
451                      .collect(Collectors.toList())));
452    }
453  }

526  public GeoShapeQueryBuilder buildGeoShapeQuery(String wkt) {
527    Geometry geometry;
528    try {
529      geometry = new WKTReader().read(wkt);
530    } catch (ParseException e) {
531      throw new IllegalArgumentException(e.getMessage(), e);
532    }
533
534    Function<Polygon, PolygonBuilder> polygonToBuilder =
535        polygon -> {
536          PolygonBuilder polygonBuilder =
537              new PolygonBuilder(
538                  new CoordinatesBuilder()
539                      .coordinates(
540                          normalizePolygonCoordinates(polygon.getExteriorRing().getCoordinates())));
541          for (int i = 0; i < polygon.getNumInteriorRing(); i++) {
542            polygonBuilder.hole(
543                new LineStringBuilder(
544                    new CoordinatesBuilder()
545                        .coordinates(
546                            normalizePolygonCoordinates(
547                                polygon.getInteriorRingN(i).getCoordinates()))));
548          }
549          return polygonBuilder;
550        };
551
552    String type =
553        "LinearRing".equals(geometry.getGeometryType())
554            ? "LINESTRING"
555            : geometry.getGeometryType().toUpperCase();
556
557    ShapeBuilder shapeBuilder = null;
558    if (("POINT").equals(type)) {
559      shapeBuilder = new PointBuilder(geometry.getCoordinate().x, geometry.getCoordinate().y);
560    } else if ("LINESTRING".equals(type)) {
561      shapeBuilder = new LineStringBuilder(Arrays.asList(geometry.getCoordinates()));
562    } else if ("POLYGON".equals(type)) {
563      shapeBuilder = polygonToBuilder.apply((Polygon) geometry);
564    } else if ("MULTIPOLYGON".equals(type)) {
565      // multipolygon
566      MultiPolygonBuilder multiPolygonBuilder = new MultiPolygonBuilder();
567      for (int i = 0; i < geometry.getNumGeometries(); i++) {
568        multiPolygonBuilder.polygon(polygonToBuilder.apply((Polygon) geometry.getGeometryN(i)));
569      }
570      shapeBuilder = multiPolygonBuilder;
571    } else {
572      throw new IllegalArgumentException(type + " shape is not supported");
573    }
574
575    try {
576      return QueryBuilders.geoShapeQuery(
577              esFieldMapper.getGeoShapeField(), shapeBuilder.buildGeometry())
578          .relation(ShapeRelation.WITHIN);
579    } catch (IOException e) {
580      throw new IllegalStateException(e.getMessage(), e);
581    }
582  }

584  /** Eliminates consecutive duplicates. The order is preserved. */
585  static Coordinate[] normalizePolygonCoordinates(Coordinate[] coordinates) {
586    List<Coordinate> normalizedCoordinates = new ArrayList<>();
587
588    // we always have to keep the fist and last coordinates
589    int i = 0;
590    normalizedCoordinates.add(i++, coordinates[0]);
591
592    for (int j = 1; j < coordinates.length; j++) {
593      if (!coordinates[j - 1].equals(coordinates[j])) {
594        normalizedCoordinates.add(i++, coordinates[j]);
595      }
596    }
597
598    return normalizedCoordinates.toArray(new Coordinate[0]);
599  }

646  private void visit(Object object, BoolQueryBuilder queryBuilder) throws QueryBuildingException {
647    Method method;
648    try {
649      method = getClass().getMethod("visit", object.getClass(), BoolQueryBuilder.class);
650    } catch (NoSuchMethodException e) {
651      log.warn(
652          "Visit method could not be found. That means a unknown Predicate has been passed", e);
653      throw new IllegalArgumentException("Unknown Predicate", e);
654    }
655    try {
656      method.invoke(this, object, queryBuilder);
657    } catch (IllegalAccessException e) {
658      log.error(
659          "This error shouldn't occur if all visit methods are public. Probably a programming error",
660          e);
661      throw new RuntimeException(e);
